Algoritmo algoritmo_colaborativo
	Definir Num Como Entero;
	Escribir ("elije que hacer:");

	Escribir ("opcion 2: Piramide") 
	Escribir ("opcion 3: Calculadora") 
	Leer Num;
	si Num=1 Entonces
		Definir N Como Real;
		Escribir ("Ingresa el numero:");
		leer N;
		f<-0;
		nor<-0;
		na<-1
		Mientras f<=N Hacer
			escribir "                   ", f
			f=nor+na
			na=nor
			nor=f
		Fin Mientras 
	FinSi
	si Num=2 Entonces
		Definir n,i,j,aux Como Entero;
		
		escribir sin saltar "ingrese el valor de n:"; Leer n;
		
		Para i<-1 hasta n con paso 1 Hacer
			para j<-1 hasta n-i con paso 1 hacer
				escribir sin saltar "  ";
			FinPara
			
			aux<-i;
			para j<-1 hasta i con paso 1 Hacer
				escribir sin saltar aux, " ";
				aux<-aux+1;
			FinPara
			
			aux<-aux-2;
			si i>=2 Entonces
				para j<-1 hasta i-1 con paso 1 Hacer
					escribir sin saltar aux," ";
					aux<-aux-1;
				FinPara
			FinSi
			
			
			escribir "";
			
			
		FinPara
		
		
	FinSi
	si Num=3 Entonces
		
		
		//declarar variables

		escribir "ingreses un numero"

			1:
				resultado=n1+n2

				resultado=n1-n2
				escribir "la resta es ",resultado

				resultado=n1*n2
				escribir "la multiplicacion es ",resultado
			4:
				resultado=n1/n2
				escribir "la multiplicacion es ", resultado
				
			De Otro Modo:
